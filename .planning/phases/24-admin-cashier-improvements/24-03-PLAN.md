def _ensure_categories_sheet(db):
    """Create Categories sheet seeded with defaults if it does not exist."""
    try:
        return db.worksheet('Categories')
    except Exception:
        sheet = db.add_worksheet(title='Categories', rows=100, cols=2)
        sheet.update([['Name'], ['Food'], ['Drinks'], ['Snacks'], ['Other']])
        return sheet


@app.route('/api/categories', methods=['GET'])
@admin_only
def get_categories():
    try:
        db = get_sheets_client()
        sheet = _ensure_categories_sheet(db)
        records = sheet.get_all_records()
        categories = [r['Name'] for r in records if r.get('Name')]
        return jsonify({'success': True, 'categories': categories})
    except Exception as e:
        logger.error(f'get_categories error: {e}', exc_info=True)
        return jsonify({'success': False, 'error': 'Failed to load categories'}), 500


@app.route('/api/categories', methods=['POST'])
@admin_only
def create_category():
    data = request.get_json() or {}
    name = (data.get('name') or '').strip()
    if not name:
        return jsonify({'success': False, 'error': 'Category name required'}), 400
    try:
        db = get_sheets_client()
        sheet = _ensure_categories_sheet(db)
        records = sheet.get_all_records()
        existing = [r['Name'].lower() for r in records if r.get('Name')]
        if name.lower() in existing:
            return jsonify({'success': False, 'error': 'Category already exists'}), 409
        sheet.append_row([name])
        return jsonify({'success': True, 'name': name})
    except Exception as e:
        logger.error(f'create_category error: {e}', exc_info=True)
        return jsonify({'success': False, 'error': 'Failed to create category'}), 500


@app.route('/api/categories/<category_name>', methods=['DELETE'])
@admin_only
def delete_category(category_name):
    try:
        db = get_sheets_client()
        # Safety check: ensure no active products use this category
        products_sheet = db.worksheet('Products')
        products = products_sheet.get_all_records()
        in_use = any(
            str(p.get('Category', '')).lower() == category_name.lower()
            and str(p.get('Active', '')).upper() == 'TRUE'
            for p in products
        )
        if in_use:
            return jsonify({
                'success': False,
                'error': f'Cannot delete "{category_name}": active products use this category'
            }), 409
        
        cat_sheet = _ensure_categories_sheet(db)
        records = cat_sheet.get_all_records()
        row_index = None
        for i, row in enumerate(records):
            if row.get('Name', '').lower() == category_name.lower():
                row_index = i + 2  # +1 header, +1 1-based
                break
        
        if row_index is None:
            return jsonify({'success': False, 'error': 'Category not found'}), 404
        
        cat_sheet.delete_rows(row_index)
        return jsonify({'success': True, 'message': f'Category "{category_name}" deleted'})
    except Exception as e:
        logger.error(f'delete_category error: {e}', exc_info=True)
        return jsonify({'success': False, 'error': 'Failed to delete category'}), 500

**3a — Remove hardcoded array:** Find `const CATEGORIES = [` and replace the whole array with:
```javascript
let CATEGORIES = [];

async function loadCategories() {
  try {
    const res = await fetch('/api/categories');
    const data = await res.json();
    if (data.success) {
      CATEGORIES = data.categories;
      populateCategorySelects();
    }
  } catch (e) {
    console.error('Failed to load categories', e);
    CATEGORIES = ['Food', 'Drinks', 'Snacks', 'Other']; // fallback
    populateCategorySelects();
  }
}

function populateCategorySelects() {
  document.querySelectorAll('select.category-select').forEach(sel => {
    const current = sel.value;
    sel.innerHTML = CATEGORIES.map(c => `<option value="${c}">${c}</option>`).join('');
    if (current) sel.value = current;
  });
}
```

function renderCategoriesList() {
  const ul = document.getElementById('categoriesList');
  if (!ul) return;
  ul.innerHTML = CATEGORIES.map(c => `
    <li class="list-group-item d-flex justify-content-between align-items-center">
      ${c}
      <button class="btn btn-sm btn-outline-danger delete-category-btn"
              data-name="${c}">Delete</button>
    </li>`).join('');
}

document.getElementById('addCategoryBtn')?.addEventListener('click', async () => {
  const input = document.getElementById('newCategoryName');
  const name = input.value.trim();
  if (!name) return;
  const res = await fetch('/api/categories', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({name})
  });
  const data = await res.json();
  if (data.success) {
    input.value = '';
    await loadCategories();
    renderCategoriesList();
  } else {
    alert(data.error || 'Failed to add category');
  }
});

document.getElementById('categoriesList')?.addEventListener('click', async (e) => {
  const btn = e.target.closest('.delete-category-btn');
  if (!btn) return;
  const name = btn.dataset.name;
  if (!confirm(`Delete category "${name}"?`)) return;
  btn.disabled = true;
  const res = await fetch(`/api/categories/${encodeURIComponent(name)}`, {method: 'DELETE'});
  const data = await res.json();
  if (data.success) {
    await loadCategories();
    renderCategoriesList();
  } else {
    alert(data.error || 'Failed to delete category');
    btn.disabled = false;
  }
});
