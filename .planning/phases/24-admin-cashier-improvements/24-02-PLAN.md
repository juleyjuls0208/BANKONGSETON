@app.route('/api/products/<product_id>', methods=['DELETE'])
@admin_only
def delete_product(product_id):
    """Soft-delete a product by setting Active=FALSE and prefixing name with [DELETED]."""
    try:
        db = get_sheets_client()
        products_sheet = db.worksheet('Products')
        records = products_sheet.get_all_records()
        
        row_index = None
        for i, row in enumerate(records):
            if str(row.get('ID', '')) == str(product_id):
                row_index = i + 2  # +1 for header, +1 for 1-based index
                break
        
        if row_index is None:
            return jsonify({'success': False, 'error': 'Product not found'}), 404
        
        # Find column indices (1-based) for Name and Active
        headers = products_sheet.row_values(1)
        name_col = headers.index('Name') + 1 if 'Name' in headers else 2
        active_col = headers.index('Active') + 1 if 'Active' in headers else 6
        
        current_name = products_sheet.cell(row_index, name_col).value or ''
        if not current_name.startswith('[DELETED]'):
            products_sheet.update_cell(row_index, name_col, f'[DELETED] {current_name}')
        products_sheet.update_cell(row_index, active_col, 'FALSE')
        
        return jsonify({'success': True, 'message': f'Product {product_id} deleted'})
    except Exception as e:
        logger.error(f'delete_product error: {e}', exc_info=True)
        return jsonify({'success': False, 'error': 'Failed to delete product'}), 500
```

document.querySelector('#productsTable tbody').addEventListener('click', function(e) {
  const btn = e.target.closest('.delete-btn');
  if (!btn) return;
  const productId = btn.dataset.id;
  const productName = btn.closest('tr').querySelector('td:first-child').textContent.trim();
  if (!confirm(`Delete "${productName}"? This cannot be undone.`)) return;
  btn.disabled = true;
  fetch(`/api/products/${productId}`, { method: 'DELETE' })
    .then(r => r.json())
    .then(data => {
      if (data.success) {
        btn.closest('tr').remove();
      } else {
        alert('Failed to delete: ' + (data.error || 'Unknown error'));
        btn.disabled = false;
      }
    })
    .catch(() => { alert('Network error'); btn.disabled = false; });
});
```
