```python
@app.route("/api/budget-summary", methods=["GET"])
@require_auth(roles=["student"])
def get_budget_summary():
    """
    Get student's month-to-date spend for the current calendar month.
    GET /api/budget-summary
    Returns: {"monthly_spend": <float>}
    """
    try:
        user_id = request.user["user_id"]

        # 1. Resolve student's money card from cache
        records = get_cached("users_all")
        if records is None:
            sheet = get_worksheet_with_retry("Students")
            records = sheet.get_all_records()
            set_cached("users_all", records, ttl=30)

        student = next(
            (r for r in records if str(r.get("StudentID", "")) == user_id), None
        )
        if not student:
            return jsonify({"error": "Student not found"}), 404

        money_card = normalize_card_uid(student.get("MoneyCardNumber", ""))

        # 2. Load transactions from cache (global cache, filter after)
        trans_sheet = get_worksheet_with_retry("Transactions Log")
        trans_records = get_cached("transactions_all")
        if trans_records is None:
            trans_records = trans_sheet.get_all_records()
            set_cached("transactions_all", trans_records, ttl=30)

        # 3. Compute current month prefix in Philippines time (e.g. "2026-03")
        now_ph = get_philippines_time()
        month_prefix = now_ph.strftime("%Y-%m")

        # 4. Sum Purchase/NFC Purchase amounts for this student this month
        monthly_spend = 0.0
        purchase_types = {"purchase", "nfc purchase"}
        for record in trans_records:
            card = normalize_card_uid(record.get("MoneyCardNumber", ""))
            if card != money_card:
                continue
            ts = record.get("Timestamp", "")
            if not ts.startswith(month_prefix):
                continue
            tx_type = record.get("TransactionType", "").lower()
            if tx_type not in purchase_types:
                continue
            try:
                monthly_spend += float(record.get("Amount", 0) or 0)
            except (ValueError, TypeError):
                pass

        return jsonify({"monthly_spend": round(monthly_spend, 2)})

    except Exception as e:
        logger.error("event=budget_summary_failed error=%s", e, exc_info=True)
        return jsonify({"error": "Internal server error"}), 500
```

Implementation notes:
- Use `purchase_types = {"purchase", "nfc purchase"}` with `.lower()` to match
  "Purchase", "NFC Purchase", etc. case-insensitively — consistent with Android
  `updateBudgetUI()` filter logic.
- `month_prefix` uses `get_philippines_time()` (already imported) to stay consistent
  with the rest of the backend. Do NOT use `datetime.utcnow()`.
- `normalize_card_uid` is already imported at line 34.
- The cache key `"transactions_all"` is global (not per-student) — this is correct;
  filtering happens in application code, identical to the existing transactions route.
