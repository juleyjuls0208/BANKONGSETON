@app.route("/api/students/<student_id>/transactions", methods=["GET"])
@login_required
def get_student_transactions(student_id):
    """Get transaction history for a specific student"""
    try:
        limit = int(request.args.get("limit", 50))
        student_id = str(student_id).strip()

        transactions_sheet = get_worksheet_with_retry("Transactions Log")
        all_transactions = transactions_sheet.get_all_records()

        # Filter by student ID (case-insensitive, strip whitespace)
        student_txns = [
            t for t in all_transactions
            if str(t.get("StudentID", "")).strip().lower() == student_id.lower()
        ]

        # Sort by Timestamp descending (most recent first)
        student_txns.sort(key=lambda t: str(t.get("Timestamp", "")), reverse=True)

        # Shape output — map Timestamp→Date, TransactionType→Type
        result = []
        for t in student_txns[:limit]:
            result.append({
                "TransactionID": t.get("TransactionID", ""),
                "Date": t.get("Timestamp", ""),
                "Type": t.get("TransactionType", ""),
                "Amount": t.get("Amount", 0),
                "BalanceBefore": t.get("BalanceBefore", 0),
                "BalanceAfter": t.get("BalanceAfter", 0),
                "Status": t.get("Status", ""),
            })

        return jsonify({
            "student_id": student_id,
            "transactions": result,
            "total": len(result)
        }), 200

    except (
        gspread.exceptions.APIError,
        gspread.exceptions.SpreadsheetNotFound,
        gspread.exceptions.WorksheetNotFound,
        ConnectionError,
        TimeoutError,
    ) as e:
        logger.error(f"Google Sheets unavailable in get_student_transactions: {e}")
        return jsonify({"error": "Service unavailable, please try again"}), 503
    except Exception as e:
        logger.error(f"Unexpected error in get_student_transactions: {e}", exc_info=True)
        return jsonify({"error": "An unexpected error occurred"}), 500
```

<script>
// Open transaction history modal for a student
async function openTxHistory(studentId, studentName) {
  document.getElementById('tx-student-label').textContent = studentName + ' (' + studentId + ')';
  document.getElementById('tx-loading').style.display = '';
  document.getElementById('tx-empty').style.display = 'none';
  document.getElementById('tx-table-wrap').style.display = 'none';

  const modal = new bootstrap.Modal(document.getElementById('tx-history-modal'));
  modal.show();

  try {
    const res = await fetch('/api/students/' + encodeURIComponent(studentId) + '/transactions?limit=50',
                            { credentials: 'same-origin' });
    const data = await res.json();
    const txns = data.transactions || [];

    document.getElementById('tx-loading').style.display = 'none';

    if (txns.length === 0) {
      document.getElementById('tx-empty').style.display = '';
      return;
    }

    const tbody = document.getElementById('tx-table-body');
    tbody.innerHTML = txns.map(t => {
      const typeClass = t.Type === 'Purchase' ? 'text-danger' :
                        t.Type === 'Load' ? 'text-success' : '';
      const amount = parseFloat(t.Amount) || 0;
      const amountStr = (amount < 0 ? '-' : '') + '₱' + Math.abs(amount).toFixed(2);
      const date = t.Date ? t.Date.slice(0, 16).replace('T', ' ') : '—';
      return `<tr>
        <td><small>${date}</small></td>
        <td><span class="${typeClass}">${t.Type || '—'}</span></td>
        <td class="${typeClass} fw-semibold">${amountStr}</td>
        <td>₱${parseFloat(t.BalanceBefore || 0).toFixed(2)}</td>
        <td>₱${parseFloat(t.BalanceAfter || 0).toFixed(2)}</td>
        <td><small>${t.Status || '—'}</small></td>
      </tr>`;
    }).join('');

    document.getElementById('tx-table-wrap').style.display = '';
  } catch (e) {
    document.getElementById('tx-loading').style.display = 'none';
    document.getElementById('tx-empty').style.display = '';
    document.getElementById('tx-empty').querySelector('p').textContent =
      'Failed to load transactions. Please try again.';
  }
}
</script>
```

// Debounce helper
function debounce(fn, ms) {
  let timer;
  return function(...args) { clearTimeout(timer); timer = setTimeout(() => fn.apply(this, args), ms); };
}

// Attach to search input
const searchInput = document.getElementById('student-search-input');

  searchInput.addEventListener('input', debounce(function() {
    loadStudents(this.value.trim());

}
