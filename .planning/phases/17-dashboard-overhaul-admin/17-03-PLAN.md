In `{% block extra_js %}`, add the export button handler:

```html
<script>
document.getElementById('export-btn').addEventListener('click', async function () {
  const start = document.getElementById('export-start').value;
  const end   = document.getElementById('export-end').value;
  const errDiv = document.getElementById('export-error');
  errDiv.style.display = 'none';

  // Build URL
  const params = new URLSearchParams();
  if (start) params.set('start_date', start);
  if (end)   params.set('end_date', end);
  const url = '/api/export/transactions' + (params.toString() ? '?' + params.toString() : '');

  // Use fetch to detect errors, then trigger download
  try {
    const res = await fetch(url, { credentials: 'same-origin' });
    if (!res.ok) {
      errDiv.style.display = '';
      return;
    }
    // Trigger download via hidden anchor
    const blob = await res.blob();
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = 'transactions' + (start ? '_' + start : '') + (end ? '_to_' + end : '') + '.csv';
    a.click();
    URL.revokeObjectURL(a.href);
  } catch (e) {
    errDiv.style.display = '';
  }
});
</script>
```
