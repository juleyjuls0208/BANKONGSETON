Create `backend/dashboard/templates/base.html` as a Jinja2 base template:
```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>{% block title %}Bangko ng Seton{% endblock %}</title>
  <!-- PWA -->
  <meta name="theme-color" content="#4F46E5">
  <link rel="manifest" href="/static/manifest.json">
  <link rel="icon" type="image/png" sizes="192x192" href="/static/icons/icon-192x192.png">
  <meta name="apple-mobile-web-app-capable" content="yes">
  <!-- Bootstrap 5 -->
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">
  <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/bootstrap-icons@1.11.0/font/bootstrap-icons.css">
  <!-- Shared styles -->
  <link rel="stylesheet" href="/static/css/pwa.css">
  <link rel="stylesheet" href="/static/css/dashboard.css">
  {% block extra_css %}{% endblock %}
</head>
<body>
  <a href="#main-content" class="skip-to-content">Skip to main content</a>

  <!-- Mobile Header -->
  <div class="mobile-header">
    <h5>Banko ng Seton</h5>
    <button class="menu-toggle" onclick="toggleSidebar()" aria-label="Toggle navigation menu">
      <i class="bi bi-list"></i>
    </button>
  </div>

  <!-- Sidebar Overlay -->
  <div class="sidebar-overlay" onclick="toggleSidebar()"></div>

  <!-- Sidebar — each page sets background-image via block -->
  <div class="sidebar" {% block sidebar_style %}{% endblock %}>
    <div class="sidebar-brand">
      Banko ng Seton
      {% if not arduino_available is defined or not arduino_available %}
      {# Web Mode badge only on dashboard where arduino_available is passed #}
      {% endif %}
    </div>
    <ul class="sidebar-menu">
      <li><a href="/dashboard" {% if active_page == 'dashboard' %}class="active"{% endif %}><i class="bi bi-speedometer2"></i> Dashboard</a></li>
      <li><a href="/students" {% if active_page == 'students' %}class="active"{% endif %}><i class="bi bi-people"></i> Students</a></li>
      <li><a href="/transactions" {% if active_page == 'transactions' %}class="active"{% endif %}><i class="bi bi-credit-card"></i> Transactions</a></li>
      {% if role == 'admin' %}
      <li><a href="/products" {% if active_page == 'products' %}class="active"{% endif %}><i class="bi bi-bag"></i> Products</a></li>
      {% endif %}
      <li><a href="/logout"><i class="bi bi-box-arrow-right"></i> Logout</a></li>
    </ul>
    <div class="mt-4 text-center" style="opacity: 0.7;">
      <small>Logged in as: {{ username }}</small>
      {% if role == 'admin' %}
      <div class="mt-2"><span class="badge bg-danger">Admin</span></div>
      {% else %}
      <div class="mt-2"><span class="badge bg-info">Finance</span></div>
      {% endif %}
    </div>
  </div>

  <!-- Main Content -->
  <div class="main-content" id="main-content">
    {% block content %}{% endblock %}
  </div>

  <!-- Bootstrap JS -->
  <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/js/bootstrap.bundle.min.js"></script>
  {% block extra_js %}{% endblock %}
  <script>
    function toggleSidebar() {
      document.querySelector('.sidebar').classList.toggle('active');
      document.querySelector('.sidebar-overlay').classList.toggle('active');
    }
  </script>
</body>
</html>
```
