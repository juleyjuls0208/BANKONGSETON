In `{% block extra_js %}`, add Chart.js CDN and the chart initialisation script:

```html
<script src="https://cdn.jsdelivr.net/npm/chart.js@4.4.0/dist/chart.umd.min.js"></script>
<script>
(function () {
  // Build last-7-days labels regardless of API data
  function getLast7Days() {
    const days = [];
    for (let i = 6; i >= 0; i--) {
      const d = new Date();
      d.setDate(d.getDate() - i);
      days.push(d.toISOString().slice(0, 10));
    }
    return days;
  }

  async function loadSpendingChart() {
    const labels = getLast7Days();
    const totals = Object.fromEntries(labels.map(d => [d, 0]));

    try {
      const res = await fetch('/api/transactions/recent?limit=500');
      if (!res.ok) throw new Error('fetch failed');
      const data = await res.json();
      const txns = data.transactions || [];

      txns.forEach(t => {
        if (!['Purchase', 'Withdrawal'].includes(t.Type)) return;
        const day = (t.Date || '').slice(0, 10);
        if (day in totals) {
          totals[day] = (totals[day] || 0) + Math.abs(parseFloat(t.Amount) || 0);
        }
      });

      document.getElementById('chart-updated').textContent =
        'Updated ' + new Date().toLocaleTimeString('en-PH', {hour: '2-digit', minute: '2-digit'});
    } catch (e) {
      // Leave all totals at 0; chart still renders with zeroes
    }

    const values = labels.map(d => totals[d]);
    const hasData = values.some(v => v > 0);

    if (!hasData) {
      document.getElementById('chart-empty').style.display = '';
      document.getElementById('spendingChart').style.display = 'none';
      return;
    }

    const ctx = document.getElementById('spendingChart').getContext('2d');
    new Chart(ctx, {
      type: 'bar',
      data: {
        labels: labels.map(d => {
          const dt = new Date(d + 'T00:00:00');
          return dt.toLocaleDateString('en-PH', {month: 'short', day: 'numeric'});
        }),
        datasets: [{
          label: 'Daily Spend (₱)',
          data: values,
          backgroundColor: 'rgba(79,70,229,0.75)',
          borderColor: 'rgba(79,70,229,1)',
          borderWidth: 1,
          borderRadius: 4
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          legend: { display: false },
          tooltip: {
            callbacks: {
              label: ctx => '₱' + ctx.parsed.y.toFixed(2)
            }
          }
        },
        scales: {
          y: {
            beginAtZero: true,
            ticks: {
              callback: v => '₱' + v.toLocaleString()
            }
          }
        }
      }
    });
  }

  document.addEventListener('DOMContentLoaded', loadSpendingChart);
})();
</script>
```
