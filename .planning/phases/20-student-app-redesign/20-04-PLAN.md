```python
@app.route("/api/student/lost-card", methods=["POST"])
def report_lost_card():
    token = request.headers.get("Authorization", "").replace("Bearer ", "")
    session = active_sessions.get(token)
    if not session:
        return jsonify({"error": "Unauthorized"}), 401

    money_card_number = session["money_card_number"]

    try:
        gc = gspread.service_account(filename=SERVICE_ACCOUNT_FILE)
        sh = gc.open_by_key(SPREADSHEET_ID)
        
        # Find the main student cards worksheet
        # Adapt worksheet name to match existing code (check what worksheet names are used)
        ws = sh.worksheet("Money Cards")  # or "Cards" or "Students" — READ THE FILE FIRST
        records = ws.get_all_records()
        headers = ws.row_values(1)
        
        # Find the Status column index (1-based)
        try:
            status_col = headers.index("Status") + 1
        except ValueError:
            # Try alternative column names
            for alt in ["status", "CardStatus", "card_status"]:
                if alt in headers:
                    status_col = headers.index(alt) + 1
                    break
            else:
                return jsonify({"error": "Status column not found in sheet"}), 500

        # Find card row
        card_row = None
        for i, row in enumerate(records, start=2):
            card_num = str(row.get("MoneyCardNumber", row.get("CardNumber", row.get("money_card_number", "")))).strip()
            if card_num == str(money_card_number).strip():
                card_row = i
                break

        if card_row is None:
            return jsonify({"error": "Card not found"}), 404

        # Check if already lost
        current_status = ws.cell(card_row, status_col).value
        if current_status == "lost":
            return jsonify({"success": True, "message": "Card already reported as lost"}), 200

        # Set status to lost
        ws.update_cell(card_row, status_col, "lost")
        
        # Invalidate the session token so the student must re-login
        # (optional but good security practice — student can still see the lost banner on next open)
        # active_sessions.pop(token, None)  # Commented out — keep session so app can show lost state

        return jsonify({"success": True, "message": "Card reported as lost"}), 200

    except Exception as e:
        return jsonify({"error": str(e)}), 500
```

```kotlin
private fun reportLostCard() {
    val token = "Bearer ${secureStorage.getToken()}"  // adapt to existing token retrieval pattern
    val btnReportLost = findViewById<MaterialButton>(R.id.btnReportLostCard)
    val tvAlreadyLost = findViewById<TextView>(R.id.tvCardAlreadyLost)

    btnReportLost.isEnabled = false
    btnReportLost.text = "Reporting..."

    lifecycleScope.launch {
        try {
            val response = apiService.reportLostCard(token)  // adapt to existing apiService variable
            if (response.isSuccessful) {
                secureStorage.setCardLost(true)
                tvAlreadyLost.visibility = View.VISIBLE
                Snackbar.make(
                    findViewById(android.R.id.content),
                    getString(R.string.lost_card_success),
                    Snackbar.LENGTH_LONG
                ).show()
            } else {
                btnReportLost.isEnabled = true
                btnReportLost.text = getString(R.string.lost_card_report_btn)
                Snackbar.make(
                    findViewById(android.R.id.content),
                    getString(R.string.lost_card_error),
                    Snackbar.LENGTH_LONG
                ).show()
            }
        } catch (e: Exception) {
            btnReportLost.isEnabled = true
            btnReportLost.text = getString(R.string.lost_card_report_btn)
            Snackbar.make(
                findViewById(android.R.id.content),
                getString(R.string.lost_card_error),
                Snackbar.LENGTH_LONG
            ).show()
        }
    }
}
```
