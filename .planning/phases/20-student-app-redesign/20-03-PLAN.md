```python
def get_or_create_budget_worksheet(sh):
    """Get the Student Budget worksheet, creating it with headers if it doesn't exist."""
    try:
        return sh.worksheet("Student Budget")
    except gspread.exceptions.WorksheetNotFound:
        ws = sh.add_worksheet(title="Student Budget", rows=1000, cols=2)
        ws.update([["MoneyCardNumber", "MonthlyLimit"]])
        return ws
```

**Step B — Add GET /api/student/budget endpoint:**

```python
@app.route("/api/student/budget", methods=["GET"])
def get_student_budget():
    token = request.headers.get("Authorization", "").replace("Bearer ", "")
    session = active_sessions.get(token)
    if not session:
        return jsonify({"error": "Unauthorized"}), 401

    money_card_number = session["money_card_number"]

    try:
        gc = gspread.service_account(filename=SERVICE_ACCOUNT_FILE)
        sh = gc.open_by_key(SPREADSHEET_ID)
        ws = get_or_create_budget_worksheet(sh)
        records = ws.get_all_records()

        monthly_limit = None
        for row in records:
            if str(row.get("MoneyCardNumber", "")).strip() == str(money_card_number).strip():
                try:
                    monthly_limit = float(row.get("MonthlyLimit", 0))
                except (ValueError, TypeError):
                    monthly_limit = 0.0
                break

        return jsonify({
            "monthly_limit": monthly_limit,  # None means not set
            "currency": "PHP"
        }), 200

    except Exception as e:
        return jsonify({"error": str(e)}), 500
```

**Step C — Add POST /api/student/budget endpoint:**

```python
@app.route("/api/student/budget", methods=["POST"])
def set_student_budget():
    token = request.headers.get("Authorization", "").replace("Bearer ", "")
    session = active_sessions.get(token)
    if not session:
        return jsonify({"error": "Unauthorized"}), 401

    money_card_number = session["money_card_number"]
    data = request.get_json()

    if not data or "monthly_limit" not in data:
        return jsonify({"error": "monthly_limit is required"}), 400

    try:
        monthly_limit = float(data["monthly_limit"])
        if monthly_limit < 0:
            return jsonify({"error": "monthly_limit must be non-negative"}), 400
    except (ValueError, TypeError):
        return jsonify({"error": "monthly_limit must be a number"}), 400

    try:
        gc = gspread.service_account(filename=SERVICE_ACCOUNT_FILE)
        sh = gc.open_by_key(SPREADSHEET_ID)
        ws = get_or_create_budget_worksheet(sh)
        records = ws.get_all_records()

        # Find existing row
        row_index = None
        for i, row in enumerate(records, start=2):  # 1-indexed, row 1 is headers
            if str(row.get("MoneyCardNumber", "")).strip() == str(money_card_number).strip():
                row_index = i
                break

        if row_index:
            ws.update_cell(row_index, 2, monthly_limit)  # Column B = MonthlyLimit
        else:
            ws.append_row([money_card_number, monthly_limit])

        return jsonify({"success": True, "monthly_limit": monthly_limit}), 200

    except Exception as e:
        return jsonify({"error": str(e)}), 500
```

```kotlin
// Add these properties to HomeActivity class:
private var monthlyLimit: Double? = null

// Add this method to HomeActivity:
private fun loadAndDisplayBudget() {
    val token = "Bearer ${secureStorage.getToken()}"  // adapt to existing token retrieval pattern
    
    lifecycleScope.launch {
        try {
            val response = apiService.getBudget(token)
            if (response.isSuccessful) {
                val budgetData = response.body()
                monthlyLimit = budgetData?.monthly_limit
                updateBudgetUI()
            }
        } catch (e: Exception) {
            // Budget card stays hidden if network error
        }
    }
}

private fun updateBudgetUI() {
    val limit = monthlyLimit
    val cardBudget = findViewById<MaterialCardView>(R.id.cardBudget)
    val budgetProgress = findViewById<CircularProgressIndicator>(R.id.budgetProgress)
    val tvBudgetPercent = findViewById<TextView>(R.id.tvBudgetPercent)
    val tvBudgetSpend = findViewById<TextView>(R.id.tvBudgetSpend)

    if (limit == null || limit <= 0) {
        cardBudget.visibility = View.VISIBLE
        tvBudgetSpend.text = getString(R.string.budget_no_limit)
        budgetProgress.progress = 0
        tvBudgetPercent.text = "--"
        return
    }

    // Compute current month spend from loaded transactions
    val currentMonth = java.time.YearMonth.now().toString()  // "YYYY-MM"
    val monthSpend = transactions  // adapt to existing transactions list variable name
        .filter { it.date.startsWith(currentMonth) }
        .sumOf { it.amount }  // adapt to existing Transaction field names

    val percent = ((monthSpend / limit) * 100).coerceIn(0.0, 100.0).toInt()
    
    cardBudget.visibility = View.VISIBLE
    budgetProgress.progress = percent
    tvBudgetPercent.text = "$percent%"
    tvBudgetSpend.text = getString(R.string.budget_spent_format, monthSpend, limit)

    // Snackbar alerts — one per threshold per month
    checkBudgetAlerts(percent, currentMonth)
}

private fun checkBudgetAlerts(percent: Int, currentMonth: String) {
    val storage = secureStorage  // adapt to existing SecureStorage variable name
    
    // Reset alert flags at month rollover
    if (storage.getBudgetAlertMonth() != currentMonth) {
        storage.setBudgetAlerted80(false)
        storage.setBudgetAlerted100(false)
        storage.setBudgetAlertMonth(currentMonth)
    }

    when {
        percent >= 100 && !storage.isBudgetAlerted100() -> {
            storage.setBudgetAlerted100(true)
            Snackbar.make(
                findViewById(android.R.id.content),
                getString(R.string.budget_alert_100),
                Snackbar.LENGTH_LONG
            ).setBackgroundTint(getColor(R.color.md_theme_light_errorContainer))
             .setTextColor(getColor(R.color.md_theme_light_onErrorContainer))
             .show()
        }
        percent in 80..99 && !storage.isBudgetAlerted80() -> {
            storage.setBudgetAlerted80(true)
            Snackbar.make(
                findViewById(android.R.id.content),
                getString(R.string.budget_alert_80),
                Snackbar.LENGTH_LONG
            ).show()
        }
    }
}
```
