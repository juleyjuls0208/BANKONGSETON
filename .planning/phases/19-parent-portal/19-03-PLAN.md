**Edit 6 — Add Set Parent JS** inside the `{% if role == 'admin' %}` script block, after the existing top-up event listener:
```javascript
// Set Parent Modal
let _setParentStudentId = null;

function openSetParentModal(studentId, studentName, currentEmail) {
  _setParentStudentId = studentId;
  document.getElementById('set-parent-student-label').textContent = studentName + ' (' + studentId + ')';
  document.getElementById('set-parent-email').value = currentEmail || '';
  document.getElementById('set-parent-password').value = '';
  document.getElementById('set-parent-feedback').style.display = 'none';
  new bootstrap.Modal(document.getElementById('setParentModal')).show();
}

document.getElementById('set-parent-confirm-btn').addEventListener('click', async function () {
  const email    = document.getElementById('set-parent-email').value.trim();
  const password = document.getElementById('set-parent-password').value;
  const feedback = document.getElementById('set-parent-feedback');
  const spinner  = document.getElementById('set-parent-spinner');
  const btnText  = document.getElementById('set-parent-btn-text');

  feedback.style.display = 'none';

  // If email provided, password is required for new accounts (allow blank = keep existing hash)
  spinner.style.display = '';
  btnText.textContent = 'Saving...';
  document.getElementById('set-parent-confirm-btn').disabled = true;

  try {
    const res = await fetch(`/api/students/${encodeURIComponent(_setParentStudentId)}/set-parent`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'same-origin',
      body: JSON.stringify({ parent_email: email, parent_password: password })
    });
    const data = await res.json();

    if (res.ok && data.success) {
      feedback.className = 'alert alert-success';
      feedback.textContent = data.message || 'Saved.';
      feedback.style.display = '';
      // Refresh student list to update badge
      loadStudents(document.getElementById('student-search-input').value.trim());
    } else {
      feedback.className = 'alert alert-danger';
      feedback.textContent = data.error || 'Failed. Please try again.';
      feedback.style.display = '';
    }
  } catch (e) {
    feedback.className = 'alert alert-danger';
    feedback.textContent = 'Network error. Please check your connection.';
    feedback.style.display = '';
  } finally {
    spinner.style.display = 'none';
    btnText.textContent = 'Save';
    document.getElementById('set-parent-confirm-btn').disabled = false;
  }
});
```
