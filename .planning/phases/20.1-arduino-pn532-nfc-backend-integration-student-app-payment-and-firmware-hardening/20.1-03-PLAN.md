**2. Add `nfc_payment` handler** (insert after `socket.on('card_read', ...)` block):
```javascript
socket.on('nfc_payment', function(data) {
    // Phone tapped — show blue NFC modal while payment is processing
    var modal = document.getElementById('paymentModal');
    var title = document.getElementById('modalTitle');
    var msg   = document.getElementById('modalMessage');
    modal.style.background = '#2196F3';
    title.textContent = 'NFC Payment';
    msg.textContent   = 'Processing payment...';
    modal.style.display = 'flex';
});
```

**3. Add `nfc_payment_result` handler** (insert immediately after the nfc_payment handler):
```javascript
socket.on('nfc_payment_result', function(data) {
    // Close the modal
    closeModal();
    var resultDiv = document.getElementById('result');
    if (data.success) {
        resultDiv.innerHTML =
            '<div style="color:#43A047;font-size:1.2em;">&#10003; NFC Payment Approved</div>';
    } else {
        var errMsg = data.error || 'Unknown error';
        resultDiv.innerHTML =
            '<div style="color:#e53935;font-size:1.1em;">&#10007; NFC Payment Failed: ' +
            errMsg +
            '</div>' +
            '<button onclick="retryNfcPayment()" ' +
            'style="margin-top:8px;padding:8px 20px;background:#2196F3;color:#fff;' +
            'border:none;border-radius:4px;cursor:pointer;">Retry NFC</button>';
    }
});
```

**4. Add `retryNfcPayment()` helper function** (add in the `<script>` block near other helpers):
```javascript
function retryNfcPayment() {
    // Re-show the NFC waiting modal so cashier can ask student to tap again
    var modal = document.getElementById('paymentModal');
    var title = document.getElementById('modalTitle');
    var msg   = document.getElementById('modalMessage');
    modal.style.background = '#2196F3';
    title.textContent = 'NFC Payment';
    msg.textContent   = 'Tap phone again...';
    modal.style.display = 'flex';
    document.getElementById('result').innerHTML = '';
}
```

**Do NOT change:** `card_read` handler, `completeSale()`, `checkout()` (except adding green color if missing), `closeModal()`, cart logic, or any CSS outside inline styles listed above.
