    from oauth2client.service_account import ServiceAccountCredentials
    import os, json

    scope = ['https://spreadsheets.google.com/feeds', 'https://www.googleapis.com/auth/drive']
    creds_path = os.path.join(os.path.dirname(__file__), '..', 'config', 'credentials.json')
    creds = ServiceAccountCredentials.from_json_keyfile_name(creds_path, scope)
    client = gspread.authorize(creds)
    sheet_name = os.getenv('GOOGLE_SHEET_NAME', 'BankongSetonDB')
    spreadsheet = client.open(sheet_name)
    users_ws = spreadsheet.worksheet('Users')
    headers = users_ws.row_values(1)
    if 'FCMToken' not in headers:
        next_col = len(headers) + 1
        users_ws.update_cell(1, next_col, 'FCMToken')
        print(f'Added FCMToken header at column {next_col}')
    else:
        print(f'FCMToken column already present at column {headers.index("FCMToken") + 1}')

if __name__ == '__main__':
    migrate_users_schema()

**Fix 1: FCM token auth mismatch (line ~730)**

Replace the `@require_auth(roles=['student'])` decorator on `register_fcm_token` with session-based auth (same pattern as `get_balance`):

```python
@app.route('/api/users/fcm-token', methods=['POST'])
def register_fcm_token():
    try:
        token = request.headers.get('Authorization', '').replace('Bearer ', '')
        if token not in active_sessions:
            return jsonify({'error': 'Invalid or expired token'}), 401
        session = active_sessions[token]
        student_id = session['student_id']
        # ... rest of FCM token storage logic uses student_id (not user_id from JWT payload)
```

Change `user_id = request.user.get('user_id')` → `student_id = session['student_id']`, update the Sheets lookup to use `student_id` instead of `user_id`.

Also fix the column letter conversion: replace `chr(64 + fcm_col_idx)` with gspread's `gspread.utils.rowcol_to_a1(user_row, fcm_col_idx)` (or equivalent: import `from gspread.utils import rowcol_to_a1` — but simplest fix is just use `users_sheet.update_cell(user_row, fcm_col_idx, fcm_token)` which avoids the letter conversion entirely).

**Fix 2: Add offset parameter to get_transactions (line ~377)**

In `get_transactions()`, change:
- `limit = int(request.args.get('limit', 50))` — keep
- Add: `offset = int(request.args.get('offset', 0))`

Change the sort+limit block from:
```python
transactions.sort(key=lambda x: x['timestamp'], reverse=True)
transactions = transactions[:limit]
return jsonify({'transactions': transactions, 'count': len(transactions)})
```
to:
```python
transactions.sort(key=lambda x: x['timestamp'], reverse=True)
total = len(transactions)
transactions = transactions[offset:offset + limit]
return jsonify({
    'transactions': transactions,
    'count': len(transactions),
    'total': total,
    'has_more': (offset + limit) < total
})
```
