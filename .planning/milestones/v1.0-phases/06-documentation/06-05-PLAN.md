```python
import os, shutil

KEEP = {
    'README.md', 'architecture.md', 'setup.md', 'api-reference.md',
    'google-sheets-schema.md', 'cashier-guide.md', 'admin-guide.md',
    'student-app.md', 'nfc-integration-guide.md'
}

os.makedirs('docs/archive', exist_ok=True)

moved = []
for fname in os.listdir('docs'):
    if fname.endswith('.md') and fname not in KEEP:
        src = os.path.join('docs', fname)
        dst = os.path.join('docs', 'archive', fname)
        shutil.move(src, dst)
        moved.append(fname)

print(f'Moved {len(moved)} files to docs/archive/')
print('Files moved:', sorted(moved))
```
