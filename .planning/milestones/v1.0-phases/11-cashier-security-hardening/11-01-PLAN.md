definitions), add:

```python
# Cashier credentials — populated at blueprint registration time (after load_dotenv)
_CASHIER_USERNAME = None
_CASHIER_PASSWORD = None

def _init_cashier_credentials(state):
    """Startup guard: abort if CASHIER_USERNAME or CASHIER_PASSWORD is missing."""
    global _CASHIER_USERNAME, _CASHIER_PASSWORD
    _CASHIER_USERNAME = os.getenv("CASHIER_USERNAME", "").strip()
    _CASHIER_PASSWORD = os.getenv("CASHIER_PASSWORD", "").strip()
    if not _CASHIER_USERNAME:
        logger.critical(
            "event=startup_aborted reason=missing_cashier_username "
            "message=\"CASHIER_USERNAME is not set. Set it in your .env file.\""
        )
        sys.exit(1)
    if not _CASHIER_PASSWORD:
        logger.critical(
            "event=startup_aborted reason=missing_cashier_password "
            "message=\"CASHIER_PASSWORD is not set. Set it in your .env file.\""
        )
        sys.exit(1)

cashier_bp.record_once(_init_cashier_credentials)
```

`sys` must be imported — add `import sys` to the imports at the top of the file if it is
not already present.
