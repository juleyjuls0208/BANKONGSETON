async function loadPorts() {
    try {
        const response = await fetch('/cashier/api/ports');
        const data = await response.json();
        const ports = data.ports || [];
        if (ports.length > 0) {
            // existing: populate port selector
            const selector = document.getElementById('portSelector');
            selector.innerHTML = '<option value="">Select COM Port</option>';

                option.textContent = port;
                selector.appendChild(option);
            });

            enterManualMode();
        }
    } catch (error) {

        enterManualMode();
    }
}

8. Add manual mode helper functions:
```javascript
function openManualLookupModal() {
    window.selectedStudent = null;
    document.getElementById('studentSearchInput').value = '';
    document.getElementById('studentSearchResults').innerHTML = '';
    document.getElementById('manualLookupModal').style.display = 'flex';
    document.getElementById('studentSearchInput').focus();
}

function closeManualModal() {
    document.getElementById('manualLookupModal').style.display = 'none';
}

async function searchStudents() {
    const q = document.getElementById('studentSearchInput').value.trim();
    if (q.length < 2) {
        document.getElementById('studentSearchResults').innerHTML = '';
        return;
    }
    try {
        const response = await fetch('/cashier/api/lookup-student?q=' + encodeURIComponent(q));
        const data = await response.json();
        const resultsDiv = document.getElementById('studentSearchResults');
        if (!data.students || data.students.length === 0) {
            resultsDiv.innerHTML = '<p style="color:#999; padding:8px;">No students found</p>';
            return;
        }
        resultsDiv.innerHTML = data.students.map(s =>
            `<div onclick="selectStudentAndPay('${s.id}','${s.name.replace(/'/g,"\\'")}',${s.balance})"
              style="padding:10px; border:1px solid #eee; border-radius:6px; margin-bottom:6px; cursor:pointer; transition:background 0.15s;"
              onmouseover="this.style.background='#f0f0f0'" onmouseout="this.style.background='white'">
              <strong>${s.name}</strong> <span style="color:#888; font-size:13px;">(${s.id})</span><br>
              <span style="color:#27ae60;">Balance: ฿${s.balance.toFixed(2)}</span>
            </div>`
        ).join('');
    } catch (error) {
        console.error('Search failed:', error);
    }
}

async function selectStudentAndPay(studentId, studentName, studentBalance) {
    closeManualModal();
    const total = cart.reduce((sum, item) => sum + (item.price * item.qty), 0);
    if (studentBalance < total) {
        alert('Insufficient funds. Balance: ฿' + studentBalance.toFixed(2) + ', Required: ฿' + total.toFixed(2));
        return;
    }
    // Show payment modal
    document.getElementById('paymentModal').style.display = 'flex';
    document.getElementById('modalTitle').textContent = 'Processing...';
    document.getElementById('modalMessage').textContent = 'Charging ' + studentName + '...';
    // Complete sale with manual_student_id
    try {
        const items = cart.map(item => ({name: item.name, price: item.price, qty: item.qty}));
        const response = await fetch('/cashier/api/complete-sale', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({manual_student_id: studentId, items: items, total: total})
        });
        const data = await response.json();
        if (data.error) {
            document.getElementById('modalTitle').textContent = 'Payment Failed';
            document.getElementById('modalMessage').textContent = data.error;
        } else {
            document.getElementById('modalTitle').textContent = 'Success!';
            document.getElementById('modalMessage').textContent =
                'Payment received!\nNew Balance: ฿' + data.new_balance.toFixed(2);
            cart = [];
            renderCart();
            setTimeout(function() { closeModal(); }, 2000);
        }
    } catch (error) {
        document.getElementById('modalTitle').textContent = 'Error';
        document.getElementById('modalMessage').textContent = error.message;
    }
}
```
